## Task 4: Jarvis Orchestrator Bot (`jarvis.py`)

**File:** `~/jarvis/jarvis.py`

This is the heart of Jarvis. Key responsibilities:
1. Telegram bot polling (python-telegram-bot 20.x)
2. `_route_message()` — calls `claude --print` to decide which agent to invoke
3. `_run_agent_quick()` — calls agent via `claude --print --system-prompt <prompt>`
4. `_spawn_agent_deep()` — creates tmux session for long-running tasks
5. All /command handlers

**Architecture of message flow:**

```
User message
    ↓
_route_message() → claude --print → JSON: {action, agent, task, mode}
    ↓
if action=="delegate" + mode=="quick":
    _run_agent_quick(agent, task) → claude --print → text output
    ↓
    Jarvis formats + replies to Telegram
    
if action=="delegate" + mode=="deep":
    _spawn_agent_deep(agent, task) → tmux session spawned
    ↓
    Jarvis sends terminal Mini App link

if action=="reply":
    Jarvis replies directly (no agent needed)
```

```python
#!/usr/bin/env python3
"""
Jarvis — Personal AI Chief of Staff
You talk to Jarvis. Jarvis routes to the right agent.
"""
import asyncio, json, logging, os, shlex, shutil, subprocess, sys, time
from pathlib import Path
from dotenv import load_dotenv

load_dotenv(os.path.join(os.path.dirname(os.path.abspath(__file__)), ".env"), override=True)
import users, store

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(name)s] %(levelname)s: %(message)s")
log = logging.getLogger("jarvis")

BOT_DIR     = Path(os.path.dirname(os.path.abspath(__file__)))
AGENTS_FILE = BOT_DIR / "agents.json"
SOUL_FILE   = BOT_DIR / "SOUL.md"
HB_FILE     = BOT_DIR / "HEARTBEAT.md"
TOKEN       = os.getenv("TELEGRAM_BOT_TOKEN", "")
OWNER_ID    = os.getenv("OWNER_ID", "")
WEBAPP_URL  = os.getenv("JARVIS_WEBAPP_URL", "")
TMUX_PREFIX = "jv_"
QUICK_TIMEOUT = 120

def _load_agents():
    return json.load(open(AGENTS_FILE)).get("agents", {}) if AGENTS_FILE.exists() else {}

def _load_soul():
    return SOUL_FILE.read_text() if SOUL_FILE.exists() else "You are Jarvis, personal AI chief of staff."

def _split(text, limit=3900):
    if not text: return ["(no output)"]
    chunks = []
    while text:
        if len(text) <= limit: chunks.append(text); break
        idx = text.rfind("\n", 0, limit) or limit
        chunks.append(text[:idx]); text = text[idx:].lstrip("\n")
    return chunks

def _tmux_exists(name):
    r = subprocess.run(["tmux", "has-session", "-t", f"{TMUX_PREFIX}{name}"], capture_output=True, timeout=5)
    return r.returncode == 0

def _tmux_list():
    try:
        r = subprocess.run(["tmux", "list-sessions", "-F", "#{session_name}"], capture_output=True, text=True, timeout=5)
        return [l[len(TMUX_PREFIX):] for l in r.stdout.splitlines() if l.startswith(TMUX_PREFIX)]
    except: return []

def _tmux_spawn(name, cmd, cwd):
    env = {**os.environ, "TERM": "xterm-256color", "COLORTERM": "truecolor"}
    subprocess.run(["tmux", "new-session", "-d", "-s", f"{TMUX_PREFIX}{name}", "-x", "220", "-y", "50", shlex.join(cmd)],
                   cwd=cwd, env=env, timeout=10, capture_output=True)

def _owner_only(handler):
    async def wrapper(self, update, context):
        if not users.is_owner(update.effective_user.id): return
        return await handler(self, update, context)
    return wrapper

def _run_agent_quick(name, cfg, task):
    cmd = ["claude", "--print", "--dangerously-skip-permissions"]
    if cfg.get("prompt"): cmd += ["--system-prompt", cfg["prompt"]]
    mcp = cfg.get("mcp_config", "")
    if mcp and os.path.exists(mcp): cmd += ["--mcp-config", mcp]
    model = cfg.get("model", "sonnet")
    cmd += ["--model", "claude-opus-4-5" if model == "opus" else "claude-sonnet-4-5"]
    cmd.append(task)
    cwd = cfg.get("cwd", os.path.expanduser("~"))
    os.makedirs(cwd, exist_ok=True)
    try:
        r = subprocess.run(cmd, capture_output=True, text=True, timeout=QUICK_TIMEOUT, cwd=cwd)
        return r.stdout.strip() or r.stderr.strip() or "(no output)"
    except subprocess.TimeoutExpired:
        return f"[{name} timed out after {QUICK_TIMEOUT}s]"
    except Exception as e:
        return f"[Error running {name}: {e}]"

def _spawn_agent_deep(name, cfg, task):
    cmd = ["claude", "--dangerously-skip-permissions"]
    if cfg.get("prompt"): cmd += ["--system-prompt", cfg["prompt"]]
    mcp = cfg.get("mcp_config", "")
    if mcp and os.path.exists(mcp): cmd += ["--mcp-config", mcp]
    cwd = cfg.get("cwd", os.path.expanduser("~"))
    os.makedirs(cwd, exist_ok=True)
    sess = f"{name}_{int(time.time()) % 10000}"
    _tmux_spawn(sess, cmd, cwd)
    return sess

def _route_message(message, agents, soul):
    agent_list = "\n".join(f"- {n}: {c['description']} | triggers: {', '.join(c.get('triggers',[]))}" for n,c in agents.items())
    prompt = f"""{soul}

## Your Team
{agent_list}

## User message
"{message}"

Respond with raw JSON only (no markdown fences):
If delegating: {{"action":"delegate","agent":"<name>","task":"<task>","mode":"quick","reason":"<why>"}}
Use mode "deep" only if task needs >2 min or interactive coding.
If answering directly: {{"action":"reply","message":"<response>"}}
Only use "reply" for greetings or things you know without research."""

    try:
        r = subprocess.run(["claude","--print","--dangerously-skip-permissions","--model","claude-sonnet-4-5", prompt],
                           capture_output=True, text=True, timeout=30)
        raw = r.stdout.strip()
        if "```" in raw: raw = raw.split("```")[1].replace("json","").strip()
        return json.loads(raw)
    except Exception as e:
        log.warning("Routing failed: %s", e)
        return {"action":"reply","message":"I had trouble routing that — could you rephrase?"}


class JarvisBot:
    def __init__(self):
        self.agents = _load_agents()
        store.sync_agents_from_json(str(AGENTS_FILE))

    @_owner_only
    async def _handle_start(self, update, context):
        from telegram import InlineKeyboardButton, InlineKeyboardMarkup
        team = ", ".join(self.agents.keys())
        text = (f"*Jarvis* — your personal AI chief of staff\n\nTeam: {team}\n\n"
                "Just message me in plain English — I'll route to the right agent.\n\n"
                "*Commands:* /hb /memory /agents /tasks /terminal /sh /status")
        kb = [[InlineKeyboardButton("Terminal", web_app={"url": WEBAPP_URL})]] if WEBAPP_URL else []
        from telegram import InlineKeyboardMarkup as IKM
        await update.message.reply_text(text, parse_mode="Markdown",
                                        reply_markup=IKM(kb) if kb else None)

    @_owner_only
    async def _handle_hb(self, update, context):
        if context.args:
            HB_FILE.write_text(" ".join(context.args))
            await update.message.reply_text("Heartbeat checklist updated.")
            return
        content = HB_FILE.read_text() if HB_FILE.exists() else "(empty)"
        for chunk in _split(f"*HEARTBEAT.md*\n\n```\n{content}\n```"):
            await update.message.reply_text(chunk, parse_mode="Markdown")

    @_owner_only
    async def _handle_memory(self, update, context):
        content = _load_soul()
        for chunk in _split(f"*SOUL.md*\n\n```\n{content}\n```"):
            await update.message.reply_text(chunk, parse_mode="Markdown")

    @_owner_only
    async def _handle_agents(self, update, context):
        self.agents = _load_agents()
        lines = ["*Jarvis Team*\n"] + [f"• *{n.capitalize()}* — {c['description']}" for n,c in self.agents.items()]
        await update.message.reply_text("\n".join(lines), parse_mode="Markdown")

    @_owner_only
    async def _handle_terminal(self, update, context):
        if not WEBAPP_URL:
            await update.message.reply_text("Terminal not available (tunnel down).")
            return
        from telegram import InlineKeyboardButton, InlineKeyboardMarkup
        await update.message.reply_text("Open terminal:",
            reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("Terminal", web_app={"url": WEBAPP_URL})]]))

    @_owner_only
    async def _handle_sh(self, update, context):
        cmd = " ".join(context.args) if context.args else ""
        if not cmd: await update.message.reply_text("Usage: /sh <command>"); return
        await context.bot.send_chat_action(chat_id=update.effective_chat.id, action="typing")
        loop = asyncio.get_running_loop()
        result = await loop.run_in_executor(None, self._run_sh, cmd)
        for chunk in _split(result):
            await update.message.reply_text(f"```\n{chunk}\n```", parse_mode="Markdown")

    def _run_sh(self, cmd):
        try:
            r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30, cwd=os.path.expanduser("~"))
            out = r.stdout + ("\n" + r.stderr if r.stderr else "")
            return (out + (f"\n[exit {r.returncode}]" if r.returncode else "")).strip() or "(no output)"
        except subprocess.TimeoutExpired: return "[Timed out]"
        except Exception as e: return f"[Error: {e}]"

    @_owner_only
    async def _handle_status(self, update, context):
        lines = ["*Jarvis Status*\n"]
        try:
            r = subprocess.run(["curl","-sf","--max-time","3",f"http://localhost:{os.getenv('JARVIS_PORT','8766')}/health"],
                               capture_output=True, timeout=5)
            lines.append(f"Web: {'OK' if r.returncode==0 else 'DOWN'}")
        except: lines.append("Web: ?")
        try:
            r = subprocess.run(["pgrep","-f","cloudflared"], capture_output=True, timeout=3)
            lines.append(f"Tunnel: {'OK' if r.returncode==0 else 'DOWN'}")
        except: lines.append("Tunnel: ?")
        sessions = _tmux_list()
        lines.append(f"Sessions: {len(sessions)}" + (f" ({', '.join(sessions)})" if sessions else ""))
        await update.message.reply_text("\n".join(lines), parse_mode="Markdown")

    @_owner_only
    async def _handle_tasks(self, update, context):
        issues = store.list_issues()
        if not issues: await update.message.reply_text("No tasks."); return
        lines = ["*Tasks*\n"] + [f"  {store.fmt_issue(i)}" for i in issues[:20]]
        await update.message.reply_text("\n".join(lines), parse_mode="Markdown")

    @_owner_only
    async def _handle_done(self, update, context):
        if not context.args: await update.message.reply_text("/done <task-id>"); return
        issue = store.get_issue(context.args[0])
        if not issue: await update.message.reply_text("Task not found."); return
        store.update_issue(issue["id"], status="done")
        await update.message.reply_text(f"Done: {issue['identifier']} — {issue['title']}")

    @_owner_only
    async def _handle_message(self, update, context):
        text = update.message.text or ""
        if not text: return
        await context.bot.send_chat_action(chat_id=update.effective_chat.id, action="typing")
        self.agents = _load_agents()
        loop = asyncio.get_running_loop()
        routing = await loop.run_in_executor(None, _route_message, text, self.agents, _load_soul())
        action = routing.get("action", "reply")

        if action == "reply":
            for chunk in _split(routing.get("message", "Not sure how to handle that.")):
                await update.message.reply_text(chunk)

        elif action == "delegate":
            name = routing.get("agent", "")
            task = routing.get("task", text)
            mode = routing.get("mode", "quick")
            reason = routing.get("reason", "")
            cfg = self.agents.get(name)
            if not cfg:
                await update.message.reply_text(f"Agent '{name}' not found."); return

            await update.message.reply_text(f"Routing to *{name.capitalize()}*\n_{reason}_", parse_mode="Markdown")

            if mode == "quick":
                await context.bot.send_chat_action(chat_id=update.effective_chat.id, action="typing")
                result = await loop.run_in_executor(None, _run_agent_quick, name, cfg, task)
                for chunk in _split(f"*{name.capitalize()}*:\n\n{result}"):
                    await update.message.reply_text(chunk, parse_mode="Markdown")
            else:
                sess = await loop.run_in_executor(None, _spawn_agent_deep, name, cfg, task)
                msg = f"*{name.capitalize()}* running in session `{sess}`"
                await update.message.reply_text(msg, parse_mode="Markdown")
                if WEBAPP_URL:
                    from telegram import InlineKeyboardButton, InlineKeyboardMarkup
                    await update.message.reply_text("Watch live:",
                        reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("Terminal", web_app={"url": WEBAPP_URL})]]))

    def run(self):
        from telegram import Update
        from telegram.ext import Application, CommandHandler, MessageHandler, filters
        if not TOKEN: print("TELEGRAM_BOT_TOKEN not set"); sys.exit(1)
        app = Application.builder().token(TOKEN).build()
        for cmd, fn in {"start":self._handle_start,"help":self._handle_start,"hb":self._handle_hb,
                        "memory":self._handle_memory,"agents":self._handle_agents,"terminal":self._handle_terminal,
                        "sh":self._handle_sh,"status":self._handle_status,"tasks":self._handle_tasks,
                        "done":self._handle_done}.items():
            app.add_handler(CommandHandler(cmd, fn))
        app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, self._handle_message))

        print("=" * 45)
        print("  Jarvis — Personal AI Chief of Staff")
        print("=" * 45)
        print(f"  Owner:  {OWNER_ID}")
        print(f"  Agents: {', '.join(self.agents.keys())}")

        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)
        async def _poll(Update):
            await app.initialize()
            try: await app.bot.get_updates(offset=-1, timeout=0)
            except: pass
            await app.start()
            await app.updater.start_polling(allowed_updates=Update.ALL_TYPES, drop_pending_updates=True)
            print("Jarvis is live.\n")
            while True: await asyncio.sleep(1)
        try: loop.run_until_complete(_poll(Update))
        except KeyboardInterrupt: print("\nShutting down...")
        finally: loop.close()

if __name__ == "__main__":
    JarvisBot().run()
```

**Step 1: Create `heartbeat.py`**

```python
#!/usr/bin/env python3
"""
Jarvis Heartbeat — Proactive Autonomous Scheduler
Wakes every HEARTBEAT_INTERVAL_MINUTES. Reads HEARTBEAT.md.
Runs claude --print to evaluate checklist.
Sends Telegram message ONLY if something needs attention.
Silent otherwise.
"""
import asyncio, json, logging, os, subprocess
from datetime import datetime
from pathlib import Path
import requests
from dotenv import load_dotenv

load_dotenv(os.path.join(os.path.dirname(os.path.abspath(__file__)), ".env"), override=True)
log = logging.getLogger("jarvis.heartbeat")

BOT_TOKEN  = os.getenv("TELEGRAM_BOT_TOKEN", "")
OWNER_ID   = os.getenv("OWNER_ID", "")
INTERVAL   = int(os.getenv("HEARTBEAT_INTERVAL_MINUTES", "30")) * 60
ACTIVE_HRS = os.getenv("HEARTBEAT_ACTIVE_HOURS", "07:00-22:00")
BOT_DIR    = Path(os.path.dirname(os.path.abspath(__file__)))

def _is_active():
    try:
        start, end = ACTIVE_HRS.split("-")
        now = datetime.now()
        def t(s): h,m = map(int,s.split(":")); return now.replace(hour=h,minute=m,second=0)
        return t(start) <= now <= t(end)
    except: return True

def _send_tg(msg):
    if not BOT_TOKEN or not OWNER_ID: return
    try:
        requests.post(f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage",
                      data={"chat_id":OWNER_ID,"text":msg,"parse_mode":"Markdown"}, timeout=10)
    except Exception as e: log.error("TG send failed: %s", e)

def _run_check(hb_content, agents_cfg, soul):
    agent_list = "\n".join(f"- {n}: {c['description']}" for n,c in agents_cfg.items())
    prompt = f"""{soul}

## Your Team
{agent_list}

## Heartbeat Checklist
{hb_content}

## Task
You are running a scheduled heartbeat check. Review the checklist.
For each item requiring action right now, check it using your tools.

If NOTHING needs attention, respond with exactly: HEARTBEAT_OK

If something needs attention, write a concise Telegram message (max 300 words) starting with:
Jarvis Heartbeat

Be selective — only alert for genuinely actionable things."""
    try:
        agents_file = BOT_DIR / "agents.json"
        agents = json.load(open(agents_file)).get("agents",{}) if agents_file.exists() else {}
        # Use atlas MCP config for web access
        atlas_mcp = agents.get("atlas",{}).get("mcp_config","")
        cmd = ["claude","--print","--dangerously-skip-permissions","--model","claude-sonnet-4-5"]
        if atlas_mcp and os.path.exists(atlas_mcp): cmd += ["--mcp-config", atlas_mcp]
        cmd.append(prompt)
        r = subprocess.run(cmd, capture_output=True, text=True, timeout=90, cwd=str(BOT_DIR))
        return r.stdout.strip() or "HEARTBEAT_OK"
    except subprocess.TimeoutExpired: return "HEARTBEAT_OK"
    except Exception as e: log.error("Check error: %s", e); return "HEARTBEAT_OK"

async def heartbeat_loop():
    log.info("Heartbeat started — interval=%ds active_hours=%s", INTERVAL, ACTIVE_HRS)
    while True:
        await asyncio.sleep(INTERVAL)
        if not _is_active():
            log.info("Outside active hours — skipping"); continue
        log.info("Running heartbeat check...")
        hb = (BOT_DIR/"HEARTBEAT.md").read_text() if (BOT_DIR/"HEARTBEAT.md").exists() else ""
        soul = (BOT_DIR/"SOUL.md").read_text() if (BOT_DIR/"SOUL.md").exists() else ""
        agents_file = BOT_DIR / "agents.json"
        agents = json.load(open(agents_file)).get("agents",{}) if agents_file.exists() else {}
        if not hb.strip(): log.info("HEARTBEAT.md empty — skipping"); continue
        result = _run_check(hb, agents, soul)
        if result.strip() == "HEARTBEAT_OK":
            log.info("Heartbeat OK — nothing to report")
        else:
            log.info("Heartbeat alert — sending to owner")
            _send_tg(result)

def main():
    logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(name)s] %(levelname)s: %(message)s")
    asyncio.run(heartbeat_loop())

if __name__ == "__main__":
    main()
```
